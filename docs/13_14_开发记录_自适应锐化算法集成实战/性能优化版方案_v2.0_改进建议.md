### 1.1 严重问题

#### 问题1: NEON代码中的mask加载错误（第100行）

**问题代码**:
```cpp
// ❌ 错误：每个BGR像素应该共享同一个mask值
uint8x8_t mask_vals = vld1_u8(mask + i / 3);
```

**问题分析**:
- `i` 是BGR字节索引（每次+24），`i/3` 得到的是像素索引
- 但 `vld1_u8` 会加载**连续8个字节**的mask值
- 实际上应该是：8个像素对应8个mask值，但每个BGR像素的3个通道应该使用**同一个**mask值
- 当前代码会导致：
  - 像素0的B通道用mask[0]
  - 像素0的G通道用mask[1] ❌ 错误！
  - 像素0的R通道用mask[2] ❌ 错误！

**正确的实现**:
```cpp
// ✅ 方案A：加载8个mask值，然后复制3次（每个BGR通道用同一个mask）
uint8x8_t mask_pixels = vld1_u8(mask + i / 3);  // 8个像素的mask

// 将每个mask值复制到对应的3个BGR通道
// 像素0: mask[0] -> B,G,R
// 像素1: mask[1] -> B,G,R
// ...
uint8x8x3_t mask_bgr;
mask_bgr.val[0] = mask_pixels;  // B通道
mask_bgr.val[1] = mask_pixels;  // G通道
mask_bgr.val[2] = mask_pixels;  // R通道

// 然后分别处理每个通道
// （需要重新设计NEON循环结构）
```

**或者更简单的方案**:
```cpp
// ✅ 方案B：按像素处理，而不是按字节处理
// 每次处理8个像素（而不是24字节）
for (; i <= total_pixels - 8; i += 8) {
    // 加载8个像素的BGR数据（使用vld3_u8交错加载）
    uint8x8x3_t bgr = vld3_u8(src + i * 3);
    uint8x8x3_t usm_bgr = vld3_u8(usm + i * 3);
    
    // 加载8个mask值
    uint8x8_t mask_vals = vld1_u8(mask + i);
    
    // 对每个通道分别处理（mask值相同）
    // B通道
    int16x8_t diff_b = vreinterpretq_s16_u16(vsubl_u8(usm_bgr.val[0], bgr.val[0]));
    int16x8_t mask16 = vreinterpretq_s16_u16(vmovl_u8(mask_vals));
    int16x8_t weighted_b = vmulq_s16(diff_b, mask16);
    int16x8_t shifted_b = vshrq_n_s16(weighted_b, 8);
    int16x8_t result_b = vaddq_s16(vreinterpretq_s16_u16(vmovl_u8(bgr.val[0])), shifted_b);
    bgr.val[0] = vqmovun_s16(result_b);
    
    // G通道（同样的mask）
    int16x8_t diff_g = vreinterpretq_s16_u16(vsubl_u8(usm_bgr.val[1], bgr.val[1]));
    int16x8_t weighted_g = vmulq_s16(diff_g, mask16);
    int16x8_t shifted_g = vshrq_n_s16(weighted_g, 8);
    int16x8_t result_g = vaddq_s16(vreinterpretq_s16_u16(vmovl_u8(bgr.val[1])), shifted_g);
    bgr.val[1] = vqmovun_s16(result_g);
    
    // R通道（同样的mask）
    int16x8_t diff_r = vreinterpretq_s16_u16(vsubl_u8(usm_bgr.val[2], bgr.val[2]));
    int16x8_t weighted_r = vmulq_s16(diff_r, mask16);
    int16x8_t shifted_r = vshrq_n_s16(weighted_r, 8);
    int16x8_t result_r = vaddq_s16(vreinterpretq_s16_u16(vmovl_u8(bgr.val[2])), shifted_r);
    bgr.val[2] = vqmovun_s16(result_r);
    
    // 存储结果（交错存储）
    vst3_u8(dst + i * 3, bgr);
}
```

**影响**: 这是一个**功能性bug**，会导致颜色通道混乱，必须修复！

#### 问题2: 函数签名不一致（第781行）

**问题代码**:
```cpp
// 声明（第673行）
static void blendWithMaskNEON(const uchar* src, const uchar* usm,
                              const uchar* mask, uchar* dst, int total_pixels);

// 实现（第781行）
void ImageProcessor::blendWithMaskNEON(const uchar* src, const uchar* usm,
                                       const uchar* mask, uchar* dst,
                                       int total_pixels) {
    const int total = total_pixels * 3;  // ❌ 这里假设total_pixels是像素数
    int i = 0;
    
    for (; i <= total - 24; i += 24) {  // ❌ 但循环按字节处理
        // ...
        uint8x8_t mask_vals = vld1_u8(mask + i / 3);  // ❌ 这里又除以3
    }
}
```

**问题分析**:
- 参数名是 `total_pixels`，但实际使用时混淆了"像素数"和"字节数"
- 循环索引 `i` 是字节索引，但 `total_pixels` 是像素数
- 导致逻辑混乱，容易出错

**修复方案**:
```cpp
// ✅ 明确参数含义
static void blendWithMaskNEON(const uchar* src_bgr, const uchar* usm_bgr,
                              const uchar* mask, uchar* dst_bgr, 
                              int width, int height);

void ImageProcessor::blendWithMaskNEON(const uchar* src_bgr, const uchar* usm_bgr,
                                       const uchar* mask, uchar* dst_bgr,
                                       int width, int height) {
    const int total_pixels = width * height;
    int i = 0;  // 像素索引
    
    // 使用vld3/vst3处理交错的BGR数据
    for (; i <= total_pixels - 8; i += 8) {
        // ... 方案B的代码
    }
    
    // 处理剩余像素
    for (; i < total_pixels; i++) {
        int mask_val = mask[i];
        for (int c = 0; c < 3; c++) {  // BGR三通道
            int idx = i * 3 + c;
            int src_val = src_bgr[idx];
            int usm_val = usm_bgr[idx];
            int diff = usm_val - src_val;
            int result = src_val + ((diff * mask_val) >> 8);
            dst_bgr[idx] = static_cast<uchar>(std::max(0, std::min(255, result)));
        }
    }
}
```

### 1.2 性能问题

#### 问题3: 查找最大值未使用NEON优化（第741行）

**问题代码**:
```cpp
// ❌ 标量循环查找最大值，76800次比较
uchar max_val = 0;
const int total = s_buffers.mask.rows * s_buffers.mask.cols;
const uchar* mask_data = s_buffers.mask.data;
for (int i = 0; i < total; i++) {
    if (mask_data[i] > max_val) max_val = mask_data[i];
}
```

**优化方案**:
```cpp
// ✅ 使用NEON查找最大值
uchar findMaxNEON(const uchar* data, int size) {
#ifdef __ARM_NEON
    uint8x16_t max_vec = vdupq_n_u8(0);
    int i = 0;
    
    // 每次处理16个字节
    for (; i <= size - 16; i += 16) {
        uint8x16_t vals = vld1q_u8(data + i);
        max_vec = vmaxq_u8(max_vec, vals);
    }
    
    // 归约：找到16个元素中的最大值
    uint8x8_t max_low = vmax_u8(vget_low_u8(max_vec), vget_high_u8(max_vec));
    max_low = vpmax_u8(max_low, max_low);
    max_low = vpmax_u8(max_low, max_low);
    max_low = vpmax_u8(max_low, max_low);
    uchar max_val = vget_lane_u8(max_low, 0);
    
    // 处理剩余元素
    for (; i < size; i++) {
        if (data[i] > max_val) max_val = data[i];
    }
    
    return max_val;
#else
    uchar max_val = 0;
    for (int i = 0; i < size; i++) {
        if (data[i] > max_val) max_val = data[i];
    }
    return max_val;
#endif
}

// 使用
uchar max_val = findMaxNEON(s_buffers.mask.data, total);
```

**性能提升**: 预计减少50-70%查找时间（从2ms降至0.5-1ms）

#### 问题4: LUT归一化可以用NEON优化（第748行）

**问题代码**:
```cpp
// ❌ OpenCV的LUT函数不一定使用NEON
cv::LUT(s_buffers.mask, cv::Mat(1, 256, CV_8U, lut), s_buffers.mask);
```

**优化方案**:
```cpp
// ✅ 手动NEON查表（如果OpenCV未优化）
void applyLUTNEON(uchar* data, int size, const uchar* lut) {
#ifdef __ARM_NEON
    int i = 0;
    for (; i <= size - 16; i += 16) {
        uint8x16_t vals = vld1q_u8(data + i);
        
        // NEON没有直接的查表指令，需要用vtbl
        // 但vtbl只支持8字节，需要分两次
        uint8x8_t low = vget_low_u8(vals);
        uint8x8_t high = vget_high_u8(vals);
        
        uint8x8_t lut_vec = vld1_u8(lut);
        uint8x8_t result_low = vtbl1_u8(lut_vec, low);
        uint8x8_t result_high = vtbl1_u8(lut_vec, high);
        
        vst1q_u8(data + i, vcombine_u8(result_low, result_high));
    }
    
    // 处理剩余
    for (; i < size; i++) {
        data[i] = lut[data[i]];
    }
#else
    for (int i = 0; i < size; i++) {
        data[i] = lut[data[i]];
    }
#endif
}
```

**注意**: OpenCV的LUT可能已经优化，需要实测对比。

#### 问题6: thread_local可能导致内存泄漏

**问题代码**:
```cpp
// ❌ thread_local变量在线程结束前不会释放
static thread_local AdaptiveSharpenBuffers s_buffers;
```

**问题分析**:
- 如果项目中频繁创建/销毁线程，每个线程都会保留一份缓冲区
- 320x240图像的缓冲区约占用：
  - gray: 76KB
  - lap: 152KB (CV_16S)
  - mask: 76KB
  - blurred: 230KB (BGR)
  - usm: 230KB (BGR)
  - **总计**: ~764KB/线程

**改进方案**:
```cpp
// ✅ 方案A：使用对象成员变量（推荐）
class ImageProcessor {
private:
    AdaptiveSharpenBuffers m_buffers;  // 每个ImageProcessor实例一份
    
public:
    void applyAdaptiveSharpenOpt(const cv::Mat &src, cv::Mat &dst,
                                 double amount, int threshold) {
        m_buffers.init(src.size());  // 使用成员变量
        // ...
    }
};

// ✅ 方案B：添加清理机制
class ImageProcessor {
public:
    static void releaseAdaptiveSharpenBuffers() {
        s_buffers.release();
    }
};

// 在适当时机调用（如切换模式、长时间不用）
m_processor->releaseAdaptiveSharpenBuffers();
```

```cpp
// test_data/test_adaptive_sharpen_correctness.cpp
void testNEONCorrectness() {
    // 创建简单测试数据
    cv::Mat src(10, 10, CV_8UC3, cv::Scalar(100, 100, 100));
    cv::Mat usm(10, 10, CV_8UC3, cv::Scalar(150, 150, 150));
    cv::Mat mask(10, 10, CV_8UC1, cv::Scalar(128));

    blendWithMaskScalar(src.data, usm.data, mask.data, dst_scalar.data, 10, 10);
    
    // NEON实现
    cv::Mat dst_neon(10, 10, CV_8UC3);
    blendWithMaskNEON(src.data, usm.data, mask.data, dst_neon.data, 10, 10);
    
    // 对比结果（允许±1误差）
    cv::Mat diff;
    cv::absdiff(dst_scalar, dst_neon, diff);
    double max_diff = 0;
    cv::minMaxLoc(diff, nullptr, &max_diff);
    
    if (max_diff <= 1) {
        qDebug() << "✅ NEON实现正确";
    } else {
        qDebug() << "❌ NEON实现错误，最大误差:" << max_diff;
        // 输出详细差异
        for (int i = 0; i < 10; i++) {
            for (int j = 0; j < 10; j++) {
                cv::Vec3b s = dst_scalar.at<cv::Vec3b>(i, j);
                cv::Vec3b n = dst_neon.at<cv::Vec3b>(i, j);
                if (s != n) {
                    qDebug() << QString("(%1,%2): scalar=%3,%4,%5 neon=%6,%7,%8")
                                .arg(i).arg(j)
                                .arg(s[0]).arg(s[1]).arg(s[2])
                                .arg(n[0]).arg(n[1]).arg(n[2]);
                }
            }
        }
    }
}
```

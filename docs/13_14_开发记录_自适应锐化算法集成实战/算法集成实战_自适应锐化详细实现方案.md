## 一、算法核心实现

### 1.1 基础版本（正确性优先）

```cpp
// imageprocessor.cpp
void ImageProcessor::applyAdaptiveSharpen(const cv::Mat &src, cv::Mat &dst,
                                          double amount, int threshold) {
    CV_Assert(src.type() == CV_8UC3 && src.data != dst.data);
    
    // 1. 计算边缘强度图（Laplacian）
    cv::Mat gray, lap;
    cv::cvtColor(src, gray, cv::COLOR_BGR2GRAY);
    cv::Laplacian(gray, lap, CV_32F, 1);  // 使用 float 避免溢出
    
    // 2. 计算掩码（归一化 + 阈值）
    cv::Mat mask;
    cv::convertScaleAbs(lap, mask);  // |Laplacian| → [0, 255]
    
    // 归一化到 [0, 1]
    mask.convertTo(mask, CV_32F, 1.0f / 255.0f);
    
    // 阈值处理：低于 threshold 的视为平坦区域
    if (threshold > 0) {
        float thresh_val = threshold / 255.0f;
        cv::threshold(mask, mask, thresh_val, 0, cv::THRESH_TOZERO);
        
        // 重新归一化到 [0, 1]
        double min_val, max_val;
        cv::minMaxLoc(mask, &min_val, &max_val);
        if (max_val > 0) {
            mask /= max_val;
        }
    }
    
    // 3. 计算 USM 锐化结果
    cv::Mat blurred, usm;
    cv::GaussianBlur(src, blurred, cv::Size(3, 3), 0);
    cv::addWeighted(src, 1.0 + amount, blurred, -amount, 0, usm);
    
    // 4. 按掩码混合
    // 将单通道 mask 扩展为三通道
    std::vector<cv::Mat> mask_channels(3, mask);
    cv::Mat mask_3ch;
    cv::merge(mask_channels, mask_3ch);
    
    // 浮点混合
    cv::Mat src_f, usm_f, dst_f;
    src.convertTo(src_f, CV_32FC3, 1.0f / 255.0f);
    usm.convertTo(usm_f, CV_32FC3, 1.0f / 255.0f);
    
    dst_f = src_f.mul(1.0f - mask_3ch) + usm_f.mul(mask_3ch);
    dst_f.convertTo(dst, CV_8UC3, 255.0f);
}
```

### 1.2 优化版本（性能优先）

```cpp
// imageprocessor.cpp - 优化版本
void ImageProcessor::applyAdaptiveSharpenOpt(const cv::Mat &src, cv::Mat &dst,
                                             double amount, int threshold) {
    CV_Assert(src.type() == CV_8UC3);
    
    QElapsedTimer timer;
    timer.start();
    
    // === 优化1：使用整数 Laplacian 避免浮点转换 ===
    cv::Mat gray, lap;
    cv::cvtColor(src, gray, cv::COLOR_BGR2GRAY);
    cv::Laplacian(gray, lap, CV_16S, 1);  // 使用 short 类型
    
    // 取绝对值并缩放到 [0, 255]
    cv::Mat mask;
    lap.convertTo(mask, CV_8U, 0.5, 128);  // 缩放并偏移
    
    qint64 t_laplacian = timer.restart();
    
    // === 优化2：整数阈值处理 ===
    if (threshold > 0) {
        // 直接使用整数阈值
        cv::threshold(mask, mask, threshold, 0, cv::THRESH_TOZERO);
        
        // 归一化：查找表方式
        uchar lut[256];
        uchar max_val = 0;
        for (int i = 0; i < 256; i++) {
            if (mask.data[i] > max_val) max_val = mask.data[i];
        }
        
        if (max_val > 0) {
            for (int i = 0; i < 256; i++) {
                lut[i] = static_cast<uchar>((mask.data[i] * 255) / max_val);
            }
            cv::LUT(mask, cv::Mat(1, 256, CV_8U, lut), mask);
        }
    }
    
    qint64 t_mask = timer.restart();
    
    // === 优化3：整数混合替代浮点混合 ===
    // 计算 USM
    cv::Mat blurred, usm;
    cv::GaussianBlur(src, blurred, cv::Size(3, 3), 0);
    cv::addWeighted(src, 1.0 + amount, blurred, -amount, 0, usm);
    
    qint64 t_usm = timer.restart();
    
    // 整数混合：dst = src + (usm - src) * mask / 255
    dst.create(src.size(), src.type());
    
    const int total_pixels = src.rows * src.cols;
    const uchar* mask_data = mask.data;
    const uchar* src_data = src.data;
    const uchar* usm_data = usm.data;
    uchar* dst_data = dst.data;
    
    // 手动循环优化（后续可改为 NEON）
    for (int i = 0; i < total_pixels * 3; i++) {
        int src_val = src_data[i];
        int usm_val = usm_data[i];
        int mask_val = mask_data[i / 3];  // 单通道 mask 用于三通道
        
        // 线性混合：dst = src + (usm - src) * alpha
        // alpha = mask / 255.0
        int diff = usm_val - src_val;
        int result = src_val + (diff * mask_val) / 255;
        
        // 钳位到 [0, 255]
        dst_data[i] = static_cast<uchar>(std::max(0, std::min(255, result)));
    }
    
    qint64 t_blend = timer.elapsed();
    
    qDebug() << QString("[自适应锐化] Laplacian:%1ms Mask:%2ms USM:%3ms Blend:%4ms Total:%5ms")
                .arg(t_laplacian).arg(t_mask).arg(t_usm).arg(t_blend)
                .arg(t_laplacian + t_mask + t_usm + t_blend);
}
```

        cv::imwrite(base_name + "_usm.jpg", dst_usm);

    }
}

void testPerformance() {
    cv::Mat test_img(240, 320, CV_8UC3, cv::Scalar(128, 128, 128));
    
    // 添加一些测试纹理
    cv::rectangle(test_img, cv::Point(50, 50), cv::Point(150, 150), 
                  cv::Scalar(200, 100, 50), -1);
    cv::circle(test_img, cv::Point(200, 100), 30, cv::Scalar(50, 150, 200), -1);
    
    const int iterations = 100;
    
    std::cout << "\n=== 性能测试 (100次迭代) ===" << std::endl;
    
    // USM 性能
    cv::Mat result;
    auto start = std::chrono::high_resolution_clock::now();
    for (int i = 0; i < iterations; i++) {
        ImageProcessor::applySharpen(test_img, result, 3.0, 1.5);
    }
    auto end = std::chrono::high_resolution_clock::now();
    auto usm_time = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    // 自适应锐化性能
    start = std::chrono::high_resolution_clock::now();
    for (int i = 0; i < iterations; i++) {
        ImageProcessor::applyAdaptiveSharpen(test_img, result, 1.5, 10);
    }
    end = std::chrono::high_resolution_clock::now();
    auto adaptive_time = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    // 优化版本性能
    start = std::chrono::high_resolution_clock::now();
    for (int i = 0; i < iterations; i++) {
        ImageProcessor::applyAdaptiveSharpenOpt(test_img, result, 1.5, 10);
    }
    end = std::chrono::high_resolution_clock::now();
    auto adaptive_opt_time = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
    
    std::cout << "USM 平均耗时: " << usm_time.count() / 100.0 << "ms/帧" << std::endl;
    std::cout << "自适应锐化平均耗时: " << adaptive_time.count() / 100.0 << "ms/帧" << std::endl;
    std::cout << "自适应锐化(优化)平均耗时: " << adaptive_opt_time.count() / 100.0 << "ms/帧" << std::endl;
}

int main() {
    testEffectiveness();
    testPerformance();
    return 0;
}

### 4.3 可能的问题与回答

**Q: 为什么选择Laplacian而不是Sobel？**
A: Laplacian对边缘更敏感，计算更简单。内窥镜图像噪声相对较少，Laplacian足够。如果需要更好的抗噪性，可以考虑Sobel。

**Q: 阈值参数如何确定？**
A: 通过实验确定。测试了多张内窥镜图像，发现阈值10/255能在保留真实边缘的同时过滤掉大部分噪声。

**Q: 有没有考虑在Lab空间做锐化？**
A: 考虑过。Lab空间只锐化L通道效果更好，但需要两次颜色转换，增加20ms开销。在当前性能约束下，选择在BGR空间做，但用自适应方法减少颜色伪影。

**Q: 如何进一步优化？**
A: 1) 用NEON指令集优化混合循环；2) 使用查找表加速；3) 降低mask分辨率；4) 尝试其他边缘检测算子。
